    @GetMapping("/queryAllFile")
    public void queryAllFile() throws IOException {
        // 指定要遍历的目录
        String directoryPath = "C:\\Users\\lcc92\\Desktop\\C2301";

        // 创建一个 File 对象
        File directory = new File(directoryPath);

        // 检查目录是否存在
        if (!directory.exists()) {
            System.out.println("目录不存在！");
            return;
        }

        // 列出目录下的所有文件
        listFiles(directory);
    }

    public void listFiles(File directory) throws IOException {
        // 获取目录下的所有文件和子目录
        File[] files = directory.listFiles();

        // 遍历文件和子目录
        for (File file : files) {
            // 如果是文件，则打印文件路径
            if (file.isFile()) {
                System.out.println("文件: " + file.getAbsolutePath());
                uploadFile2(file, 3);
                // 上传文件
                FileUploadResponse response = uploadFile2(file,3);
                FileUploadVO fileUploadVO = BeanUtil.toBean(response.getData(), FileUploadVO.class);
                //新增学生
                saveArchives(fileUploadVO, file.getAbsoluteFile());
            }
            // 如果是目录，则递归调用 listFiles 方法
            else if (file.isDirectory()) {
                System.out.println("目录: " + file.getAbsolutePath());
                listFiles(file); // 递归调用
            }
        }
    }

     public FileUploadResponse uploadFile2(File file, Integer upType) throws IOException {
        String requestUrl = "https://xxx/singleFile";
        String boundary = Long.toHexString(System.currentTimeMillis()); // 随机分隔符
        String CRLF = "\r\n"; // 回车换行符

        // 设置请求头
        HttpURLConnection connection = (HttpURLConnection) new URL(requestUrl).openConnection();
        connection.setRequestMethod("POST");
        connection.setDoOutput(true);
        connection.setRequestProperty("Content-Type", "multipart/form-data; boundary=" + boundary);
        connection.setRequestProperty("Authorization", "Bearer eyJ0eXAiOiJKV1QiLCJhbGciOiJb3k");

        try (
                // 构建请求体
                OutputStream output = connection.getOutputStream();
                PrintWriter writer = new PrintWriter(new OutputStreamWriter(output, "UTF-8"), true);
        ) {
            // 写入文件数据
            writer.append("--" + boundary).append(CRLF);
            writer.append("Content-Disposition: form-data; name=\"file\"; filename=\"" + URLEncoder.encode(file.getName(), "UTF-8") + "\"").append(CRLF);

            writer.append("Content-Type: " + HttpURLConnection.guessContentTypeFromName(file.getName())).append(CRLF); // 根据文件名推测类型
            writer.append(CRLF).flush();
            Files.copy(file.toPath(), output);
            output.flush(); // 输出结束
            writer.append(CRLF).flush();

            // 写入upType参数
            writer.append("--" + boundary).append(CRLF);
            writer.append("Content-Disposition: form-data; name=\"upType\"").append(CRLF);
            writer.append(CRLF).append(String.valueOf(upType)).append(CRLF).flush();

            // 请求结束标志
            writer.append("--" + boundary + "--").append(CRLF).flush();
        }

    # 分割文件名称

        public static String[] parseFileName(String filename) {

        // 定义正则表达式，匹配"姓名_编号"格式的字符串
        String regex = "^(.*?)_(.*?)\\..*$";

        // 创建 Pattern 对象
        Pattern pattern = Pattern.compile(regex);

        // 创建 Matcher 对象
        Matcher matcher = pattern.matcher(filename);

        // 创建存储结果的数组
        String[] result = new String[2];

        // 查找匹配的部分
        if (matcher.find()) {
            // 提取结果并存储到数组中
            result[0] = matcher.group(1); 
            result[1] = matcher.group(2); 
        } else {
            System.out.println("未找到匹配项！");
        }
        return result;
    }

    # 生成随机身份证

           public String generateIdCard() {
        String provinceCode = "110105"; // 北京
        String birthDate = generateRandomBirthDate(); // 生成随机的出生日期
        String sequenceCode = generateRandomSequenceCode(4); // 生成随机的顺序码和校验码
        // 拼接身份证号码
        String idCardNumber = provinceCode + birthDate + sequenceCode;
        return idCardNumber;
    }
    
   
   // 生成随机的出生日期（12到18岁之间）
    public static String generateRandomBirthDate() {
        Random random = new Random();
        // 生成12到18岁之间的随机年份
        int year = random.nextInt(7) + 2006; // 2024 - 12 = 2012, 2024 - 18 = 2006
        // 生成随机月份
        int month = random.nextInt(12) + 1;
        // 生成随机日
        int day = random.nextInt(28) + 1; // 简化处理，假设每个月都是28天

        // 格式化成身份证要求的日期格式
        return String.format("%04d%02d%02d", year, month, day);
    }
		//生成最后四位随机数
    public static String generateRandomSequenceCode(int length) {
        String chars = "0123456789";
        SecureRandom random = new SecureRandom();
        StringBuilder sb = new StringBuilder(length);

        for (int i = 0; i < length; i++) {
            int randomIndex = random.nextInt(chars.length());
            char randomChar = chars.charAt(randomIndex);
            sb.append(randomChar);
        }
        return sb.toString();
    }
